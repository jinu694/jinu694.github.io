~~~python
weight=int(input('본인의 몸무게를 입력하시오'))
if weight<63.5:
   print('Light급')
elif weight<67:
  print('welter급')
elif weight<71:
  print('Lighter welter 급')
elif weight<75:
  print('Middle 급')
elif weight<81:
  print('Light heavy 급')
else:
  print('heavy급')
~~~
